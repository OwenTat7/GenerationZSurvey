import matplotlib.pyplot as plt

# Count the number of people in each employment status
status_counts = employment_dummies.sum().sort_values(ascending=False)

# Plot as a bar chart
bars = plt.bar(status_counts.index, status_counts.values, color='skyblue', edgecolor='black')

# Add value labels on top of each bar
for bar in bars:
    height = bar.get_height()
    plt.text(
        bar.get_x() + bar.get_width() / 2,  # x position
        height,                             # y position
        str(int(height)),                   # label text
        ha='center', va='bottom', fontsize=9
    )

# Label the plot
plt.title('Employment Status Distribution')

plt.ylabel('Number of Respondents')
plt.xticks(rotation=45)
plt.tight_layout()

plt.show()

import matplotlib.pyplot as plt

# Define custom order for education levels
custom_order = [
    "High school",
    "Some college",
    "Associate degree",
    "Bachelor’s degree",
    "Graduate degree"
]

# Get value counts and reindex
edu_counts = df['Education'].value_counts().reindex(custom_order)

# Create bar chart
plt.bar(edu_counts.index, edu_counts.values, color='lightcoral', edgecolor='black')

# Add value labels
for i, v in enumerate(edu_counts.values):
    if pd.notna(v):  # avoid placing text on missing bars
        plt.text(i, v + .2, str(int(v)), ha='center')

# Add labels and title
plt.title('Education Level (Sorted)')
plt.ylabel('Frequency')
plt.xticks(rotation=45)

plt.tight_layout()
plt.show()
